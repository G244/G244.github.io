触发器

-- 触发器实时更新聚合函数数据
drop trigger if EXISTS user_trigger;
DELIMITER $$
create trigger user_trigger  AFTER INSERT ON qsdk_user for each row 
BEGIN
      set @old_totalCount = 0;
      set @old_productSum = 0;
      select 
        IFNULL(totalCount,0),iFNULL(productSum,0)
      from  qsdk_user_view
      into
        @old_totalCount,@old_productSum;
      set @new_totalCount =@old_totalCount+1;
      set @new_productSum =@old_productSum+NEW.productId;
      update
      qsdk_user_view set totalCount=@new_totalCount,productSum = @new_productSum;
END;
$$
DELIMITER;

储存过程

-- 存储过程聚合函数数据(需用定时任务)
drop PROCEDURE if EXISTS user_procedure;
DELIMITER $$
CREATE PROCEDURE user_procedure()
BEGIN
  TRUNCATE table qsdk_user_view;
  insert into qsdk_user_view select * from user_view;
END;
$$
DELIMITER;
call user_procedure();

　存储引擎的区别

　　　　　　　　　　 　　　

　　　　InnoDB：支持事务处理，支持外键，支持崩溃修复能力和并发控制。如果需要对事务的完整性要求比较高（比如银行）
　　　　，要求实现并发控制（比如售票），那选择InnoDB有很大的优势。如果需要频繁的更新、删除操作的数据库
　　　　MyISAM：插入数据快，空间和内存使用比较低。如果表主要是用于插入新记录和读出记录，那么选择MyISAM能实现处理高效率。

字段类型

（2）char的存储方式是，对英文字符（ASCII）占用1个字节，对一个汉字占用两个字节。长度不可变

（3）varchar的存储方式是，对每个英文字符占用2个字节，汉字也占用2个字节。长度可变

5.7以下超过最大值的数将存最大值,大于5.7存 大于最大值将报错
bigint -2^63 2^63-1
int -2^31 2^31-1
smallint -2^15(-32768) 2^15-1(32767)
tinyint -2^7（128） 2^7-1 （127）
如果字段设置为无符号时,值将为负数的绝对值+正数值

用户管理

  • 创建用户
  • create user ‘用户名‘@’ip’ identified by ‘密码’
    • 同时创建用户赋予权限
      grant select,update,insert on databases.* to ‘用户名‘@’%’ identified by ‘密码’
    • 赋予指定权限
      grant select,update,insert on databases.* to ‘用户名‘@’%’;
    • 查看权限
      show grants for 用户名
    • 赋予全部权限
      grant all privileges on . to ‘用户名‘@’ip’;
    • 删除权限
      revoke drop,insert on databases.* from ‘用户名‘@’%’;
    • 修改生效
      FLUSH PRIVILEGES;
